2   import java.util.Comparator;

4   /**
5    * Represents Sort By Year, an implementation of Comparator for Media
6    * so I can sort media by their release date from most recent to least
7    * recent.
8    *
9    * @author Goch.
10   */
11  public class SortByYear implements Comparator<Media> {
12  
13    /**
14     * Compares media a to media b and returns a -1 if media b has a more
15     * recent release date than media a.
16     * @param a the first media we are comparing.
17     * @param b the second media we are comparing.
18     * @return -1 if the b was released after a, a 1 if b was released before a, or a
19     * 0 if b and a were released at the same time.
20     */
21    public int compare(Media a, Media b){
22      return b.releaseYear.compareTo(a.releaseYear);
23    }
24  }
