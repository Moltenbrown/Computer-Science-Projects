7   /**
8    * Represents Abstract Media with its details--the abstract media's title as a string, the release year
9    * as a four (4) digit integer, a list of directors, a list of actors, and the amount the movie has been
10   * streamed as an integer.
11   *
12   * @author Goch.
13   */
14  public abstract class AbstractMedia {
15    String title;
16    Integer releaseYear;
17    List directors;
18    List actors;
19    Integer amountOfStreams;
20    TypeOfMedia typeOfMedia;
21  
22    /**
23     * Create abstract media from a title, release year, a list of directors, a list of actors,
24     * and the amount the movie has been streamed. A four digit exception is thrown if the user enters
25     * a release year that is anything other than four (4) digits.
26     *
27     * @param title the abstract media's title.
28     * @param releaseYear the abstract media's release year.
29     * @param directors the directors who worked on the abstract media.
30     * @param actors the main actors of the abstract media.
31     * @param amountOfStreams the amount of times the abstract media was streamed.
32     * @param typeOfMedia the type of media the abstract media is.
33     * @throws FourDigitException if the user enters a release year that is not four (4) digits.
34     */
35    public AbstractMedia(String title, Integer releaseYear, List directors, List actors,
36        Integer amountOfStreams, TypeOfMedia typeOfMedia) throws FourDigitException {
37      this.title = title;
38      this.releaseYear = releaseYear;
39      if(releaseYear.toString().length() != 4){
40        throw new FourDigitException("The release year must be four (4) digits.");
41      }
42      this.directors = directors;
43      this.actors = actors;
44      this.amountOfStreams = amountOfStreams;
45      this.typeOfMedia = typeOfMedia;
46    }
47  
48    /**
49     * Return the abstract media title.
50     * @return the abstract media title.
51     */
52    public String getTitle() {
53      return title;
54    }
55  
56    /**
57     * Return the abstract media release year.
58     * @return the abstract media release year.
59     */
60    public Integer getReleaseYear() {
61      return releaseYear;
62    }
63  
64    /**
65     * Return the abstract media release year.
66     * @return the abstract media release year.
67     */
68    public List getDirectors() {
69      return directors;
70    }
71  
72    /**
73     * Returns the amount of times the media was streamed.
74     * @return the amount of streams.
75     */
76    public Integer getAmountOfStreams() {
77      return amountOfStreams;
78    }
79  
80    /**
81     * Sets the amount of streams of the media to whatever number is entered.
82     * @param amountOfStreams the number that we want to set amount of streams to.
83     */
84    public void setAmountOfStreams(Integer amountOfStreams) {
85      this.amountOfStreams = amountOfStreams;
86    }
87  
88    /**
89     * Returns the type of media the abstract media is.
90     * @return the media type.
91     */
92  
93    public TypeOfMedia getTypeOfMedia() {
94      return typeOfMedia;
95    }
96  
97    /**
98     * Returns whether the director entered is a director of the abstract media.
99     * @param director the director you are searching for.
100    * @return true if the director is one of the directors on the abstract media,
101    * false otherwise.
102    */
103   public Boolean isDirector(Director director){ return this.directors.contains(director);}
104 
105   /**
106    * Evaluates whether the object o is the same as the abstract media.
107    * @param o the object being compared.
108    * @return true if the object is the same as the abstract media, false otherwise.
109    */
110   @Override
111   public boolean equals(Object o) {
112     if (this == o) {
113       return true;
114     }
115     if (!(o instanceof AbstractMedia)) {
116       return false;
117     }
118     AbstractMedia that = (AbstractMedia) o;
119     return Objects.equals(title, that.title) &&
120         Objects.equals(releaseYear, that.releaseYear) &&
121         Objects.equals(directors, that.directors) &&
122         Objects.equals(actors, that.actors) &&
123         Objects.equals(amountOfStreams, that.amountOfStreams) &&
124         Objects.equals(typeOfMedia, that.typeOfMedia);
125   }

130    */
131   @Override
132   public int hashCode() {
133

135   }

139    * @return a string description of the abstract media.
140    */
141   @Override
142   public String toString() {
143     return "AbstractMedia{" +
144         "title='" + title + '\'' +
145         ", releaseYear=" + releaseYear +
146         ", directors=" + directors +
147         ", actors=" + actors +
148         ", amountOfStreams=" + amountOfStreams +
149         ", mediaType=" + typeOfMedia +
