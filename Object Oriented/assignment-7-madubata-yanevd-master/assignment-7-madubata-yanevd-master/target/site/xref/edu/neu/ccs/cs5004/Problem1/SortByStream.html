2   import java.util.Comparator;
3   
4   /**
5    * Represents Sort By Stream, an implementation of Comparator for Media
6    * so I can sort media by their stream amount from highest to lowest.
7    *
8    * @author Goch.
9    */
10  public class SortByStream implements Comparator<Media> {
11  
12    /**
13     * Compares media a to media b and returns a -1 if media b was streamed more
14     * than media a.
15     * @param a the first media we are comparing.
16     * @param b the second media we are comparing.
17     * @return -1 if the b was streamed more than a, a 1 if b was streamed less
18     * than a, or a 0 if b and a were streamed the same amount..
19     */
20    public int compare(Media a, Media b){
21      return b.amountOfStreams.compareTo(a.amountOfStreams);
22    }
23  }
